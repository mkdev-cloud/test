    <script>
        class CodeScrambleGame {
            constructor() {
                this.currentDifficulty = 'easy';
                this.score = 0;
                this.streak = 0;
                this.timeLeft = 30;
                this.gameActive = false;
                this.currentQuestion = null;
                this.timer = null;
                
                // Bug database with different difficulty levels
                this.bugDatabase = {
                    easy: [
                        {
                            language: 'Python',
                            bugType: 'Syntax Error',
                            description: 'Missing colon after if statement',
                            buggyCode: `# Check if number is positive
def check_positive(num):
    if num > 0
        return "Positive"
    else:
        return "Not positive"`,
                            correctCode: `# Check if number is positive
def check_positive(num):
    if num > 0:
        return "Positive"
    else:
        return "Not positive"`,
                            hint: 'Python if statements need a colon (:) at the end'
                        },
                        {
                            language: 'JavaScript',
                            bugType: 'Logic Error',
                            description: 'Wrong comparison operator',
                            buggyCode: `// Check if user is old enough to vote
function canVote(age) {
    if (age > 18) {
        return true;
    }
    return false;
}`,
                            correctCode: `// Check if user is old enough to vote
function canVote(age) {
    if (age >= 18) {
        return true;
    }
    return false;
}`,
                            hint: 'Should people exactly 18 years old be able to vote?'
                        },
                        {
                            language: 'Python',
                            bugType: 'Indentation Error',
                            description: 'Incorrect indentation in function',
                            buggyCode: `def greet_user(name):
print(f"Hello, {name}!")
print("Welcome to our program!")`,
                            correctCode: `def greet_user(name):
    print(f"Hello, {name}!")
    print("Welcome to our program!")`,
                            hint: 'Python requires proper indentation for function bodies'
                        }
                    ],
                    medium: [
                        {
                            language: 'Python',
                            bugType: 'Logic Error',
                            description: 'Off-by-one error in loop',
                            buggyCode: `# Print numbers 1 to 10
def print_numbers():
    for i in range(10):
        print(i + 1)`,
                            correctCode: `# Print numbers 1 to 10
def print_numbers():
    for i in range(1, 11):
        print(i)`,
                            hint: 'Check the range parameters for printing 1 to 10'
                        },
                        {
                            language: 'JavaScript',
                            bugType: 'Type Error',
                            description: 'String concatenation instead of addition',
                            buggyCode: `// Calculate total price with tax
function calculateTotal(price, taxRate) {
    let tax = price * taxRate;
    let total = price + tax;
    return "Total: $" + total;
}

// This will cause issues with math
let result = calculateTotal("100", 0.08);`,
                            correctCode: `// Calculate total price with tax
function calculateTotal(price, taxRate) {
    price = parseFloat(price);
    let tax = price * taxRate;
    let total = price + tax;
    return "Total: $" + total.toFixed(2);
}

// This will cause issues with math
let result = calculateTotal("100", 0.08);`,
                            hint: 'String inputs need to be converted to numbers for math operations'
                        },
                        {
                            language: 'Python',
                            bugType: 'List Index Error',
                            description: 'Accessing list index that may not exist',
                            buggyCode: `def get_first_last(items):
    first = items[0]
    last = items[-1]
    return first, last

# This will crash with empty list
result = get_first_last([])`,
                            correctCode: `def get_first_last(items):
    if len(items) == 0:
        return None, None
    first = items[0]
    last = items[-1]
    return first, last

# This will crash with empty list
result = get_first_last([])`,
                            hint: 'What happens when you try to access the first item of an empty list?'
                        }
                    ],
                    hard: [
                        {
                            language: 'JavaScript',
                            bugType: 'Closure Issue',
                            description: 'Variable scope problem in loop',
                            buggyCode: `// Create array of functions
function createFunctions() {
    let functions = [];
    for (var i = 0; i < 3; i++) {
        functions.push(function() {
            return i;
        });
    }
    return functions;
}

// All functions return 3 instead of 0, 1, 2`,
                            correctCode: `// Create array of functions
function createFunctions() {
    let functions = [];
    for (let i = 0; i < 3; i++) {
        functions.push(function() {
            return i;
        });
    }
    return functions;
}

// All functions return 3 instead of 0, 1, 2`,
                            hint: 'The issue is with variable scoping - use let instead of var'
                        },
                        {
                            language: 'Python',
                            bugType: 'Mutable Default Argument',
                            description: 'Dangerous mutable default parameter',
                            buggyCode: `def add_item(item, shopping_list=[]):
    shopping_list.append(item)
    return shopping_list

# This causes unexpected behavior
list1 = add_item("apples")
list2 = add_item("bananas")
# list2 now contains both apples and bananas!`,
                            correctCode: `def add_item(item, shopping_list=None):
    if shopping_list is None:
        shopping_list = []
    shopping_list.append(item)
    return shopping_list

# This causes unexpected behavior
list1 = add_item("apples")
list2 = add_item("bananas")
# list2 now contains both apples and bananas!`,
                            hint: 'Mutable default arguments are shared between function calls'
                        },
                        {
                            language: 'JavaScript',
                            bugType: 'Async/Promise Error',
                            description: 'Missing await in async function',
                            buggyCode: `async function fetchUserData(userId) {
    try {
        let response = fetch(\`/api/users/\${userId}\`);
        let userData = response.json();
        return userData.name;
    } catch (error) {
        return "Error loading user";
    }
}`,
                            correctCode: `async function fetchUserData(userId) {
    try {
        let response = await fetch(\`/api/users/\${userId}\`);
        let userData = await response.json();
        return userData.name;
    } catch (error) {
        return "Error loading user";
    }
}`,
                            hint: 'fetch() and .json() return promises that need to be awaited'
                        }
                    ]
                };
                
                this.initializeEventListeners();
            }
            
            initializeEventListeners() {
                // Difficulty selection
                document.querySelectorAll('.difficulty-btn').forEach(btn => {
                    btn.addEventListener('click', (e) => {
                        document.querySelectorAll('.difficulty-btn').forEach(b => b.classList.remove('active'));
                        e.target.classList.add('active');
                        this.currentDifficulty = e.target.dataset.difficulty;
                    });
                });
                
                // Game controls
                document.getElementById('start-game').addEventListener('click', () => this.startGame());
                document.getElementById('check-answer').addEventListener('click', () => this.checkAnswer());
                document.getElementById('skip-question').addEventListener('click', () => this.skipQuestion());
                
                // Enter key to submit answer
                document.getElementById('answer-input').addEventListener('keydown', (e) => {
                    if (e.key === 'Enter' && e.ctrlKey) {
                        this.checkAnswer();
                    }
                });
            }
            
            startGame() {
                this.gameActive = true;
                this.score = 0;
                this.streak = 0;
                this.timeLeft = 30;
                
                document.getElementById('game-area').classList.remove('hidden');
                document.getElementById('start-game').textContent = 'New Game';
                
                this.loadNewQuestion();
                this.startTimer();
                this.updateDisplay();
            }
            
            loadNewQuestion() {
                const questions = this.bugDatabase[this.currentDifficulty];
                this.currentQuestion = questions[Math.floor(Math.random() * questions.length)];
                
                document.getElementById('language-tag').textContent = this.currentQuestion.language;
                document.getElementById('bug-type').textContent = this.currentQuestion.bugType;
                document.getElementById('bug-description').textContent = this.currentQuestion.description;
                document.getElementById('buggy-code').textContent = this.currentQuestion.buggyCode;
                document.getElementById('answer-input').value = '';
                document.getElementById('feedback').classList.add('hidden');
            }
            
            checkAnswer() {
                if (!this.gameActive || !this.currentQuestion) return;
                
                const userAnswer = document.getElementById('answer-input').value.trim();
                const correctAnswer = this.currentQuestion.correctCode.trim();
                
                // Simple comparison - in a real game, you'd want more sophisticated checking
                const isCorrect = this.compareCode(userAnswer, correctAnswer);
                
                const feedback = document.getElementById('feedback');
                feedback.classList.remove('hidden', 'correct', 'incorrect');
                
                if (isCorrect) {
                    feedback.classList.add('correct');
                    feedback.textContent = '🎉 Correct! Great debugging skills!';
                    this.score += this.getDifficultyPoints();
                    this.streak++;
                    this.timeLeft += 5; // Bonus time for correct answer
                } else {
                    feedback.classList.add('incorrect');
                    feedback.innerHTML = `❌ Not quite right. Hint: ${this.currentQuestion.hint}<br><br>Expected solution:<br><code>${this.currentQuestion.correctCode}</code>`;
                    this.streak = 0;
                }
                
                this.updateDisplay();
                setTimeout(() => this.loadNewQuestion(), 3000);
            }
            
            skipQuestion() {
                if (!this.gameActive) return;
                
                const feedback = document.getElementById('feedback');
                feedback.classList.remove('hidden', 'correct', 'incorrect');
                feedback.classList.add('incorrect');
                feedback.innerHTML = `⏭️ Skipped! Here's the solution:<br><code>${this.currentQuestion.correctCode}</code><br>Hint: ${this.currentQuestion.hint}`;
                
                this.streak = 0;
                this.updateDisplay();
                setTimeout(() => this.loadNewQuestion(), 3000);
            }
            
            compareCode(user, correct) {
                // Remove whitespace and compare
                const normalize = (code) => code.replace(/\s+/g, ' ').toLowerCase().trim();
                return normalize(user) === normalize(correct);
            }
            
            getDifficultyPoints() {
                const points = { easy: 10, medium: 20, hard: 30 };
                return points[this.currentDifficulty] + (this.streak * 5);
            }
            
            startTimer() {
                this.timer = setInterval(() => {
                    this.timeLeft--;
                    document.getElementById('timer').textContent = this.timeLeft;
                    
                    if (this.timeLeft <= 10) {
                        document.getElementById('timer').classList.add('pulse');
                    }
                    
                    if (this.timeLeft <= 0) {
                        this.endGame();
                    }
                }, 1000);
            }
            
            endGame() {
                this.gameActive = false;
                clearInterval(this.timer);
                
                const feedback = document.getElementById('feedback');
                feedback.classList.remove('hidden', 'correct', 'incorrect');
                feedback.innerHTML = `🏁 Game Over! Final Score: ${this.score}<br>Great job debugging! Click "New Game" to play again.`;
                
                document.getElementById('timer').classList.remove('pulse');
            }
            
            updateDisplay() {
                document.getElementById('current-score').textContent = this.score;
                document.getElementById('current-streak').textContent = this.streak;
            }
        }
        
        // Initialize the game when page loads
        document.addEventListener('DOMContentLoaded', () => {
            new CodeScrambleGame();
        });
    </script>
